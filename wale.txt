The core pattern for every rule:

type RuleResult = {
  triggered: boolean
  confidence: 'high' | 'medium' | 'low' | 'excluded'
  evidence: string[]
  billing: string[]
  time_sensitive?: boolean
  next_unlock?: string
}

// Date windows expressed as explicit boolean arithmetic — never string comparison
function cecWindowCheck(answers: Answers, asOf: Date): RuleResult {
  const endDate = answers.canadian_work_end_date
    ? new Date(answers.canadian_work_end_date)
    : new Date() // ongoing job

  const windowStart = subYears(asOf, 3)
  const hoursInWindow = computeHoursInWindow(answers.canadian_work_history, windowStart, asOf)

  if (hoursInWindow >= 1560) {
    return { triggered: true, confidence: 'high', evidence: [`${hoursInWindow} hrs in 3-yr window`], billing: ['B-02'] }
  }

  if (hoursInWindow > 0) {
    const monthsToThreshold = estimateMonthsToThreshold(hoursInWindow, answers.weekly_hours)
    return {
      triggered: false, confidence: 'low',
      evidence: [`${hoursInWindow} hrs — ${monthsToThreshold} months to CEC`],
      billing: ['B-07'],
      next_unlock: `${monthsToThreshold} months to CEC threshold`
    }
  }

  return { triggered: false, confidence: 'excluded', evidence: ['No qualifying Canadian work'], billing: [] }
}
Confidence aggregation when multiple results trigger:

// Rule: aggregate = min(individual confidences), with a floor adjustment if 3+ high results
const CONFIDENCE_WEIGHT = { high: 1.0, medium: 0.6, low: 0.3, excluded: 0.0 }

function aggregateConfidence(results: RuleResult[]): number {
  const triggered = results.filter(r => r.triggered)
  if (!triggered.length) return 0
  const weights = triggered.map(r => CONFIDENCE_WEIGHT[r.confidence])
  return Math.min(...weights) // pessimistic — safest for outreach
}

// Outreach tier from aggregate:
// >= 0.8 → 'high' (safe to send personalised message with specific pathway)
// >= 0.5 → 'medium' (soft outreach: "your situation may have changed, let's review")
// < 0.5  → 'low' (consultant validation required before any outreach)
CRS calculator — deterministic, no LLM, no estimation:

function calculateCRS(answers: Answers): { score: number, breakdown: Record<string, number> } {
  // Core human capital factors
  const age       = scoreAge(answers.date_of_birth)
  const education = scoreEducation(answers.education_level, answers.has_eca)
  const language  = scoreLanguage(answers.primary_language_scores, answers.secondary_language_scores)
  const canadianWork = scoreCdnWork(answers.canadian_work_years)
  const foreignWork  = scoreForeignWork(answers.foreign_work_years_in_last5)

  // Spouse/partner factors (if applicable)
  const spouse = answers.is_married
    ? scoreSpouse(answers.spouse_language, answers.spouse_education, answers.spouse_cdn_work)
    : 0

  // Additional factors
  const sibling  = answers.sibling_in_canada ? 15 : 0
  const jobOffer = scoreJobOffer(answers.job_offer_lmia, answers.job_offer_teer)
  const pnp      = answers.has_pnp_nomination ? 600 : 0

  const score = age + education + language + canadianWork + foreignWork + spouse + sibling + jobOffer + pnp

  return { score, breakdown: { age, education, language, canadianWork, foreignWork, spouse, sibling, jobOffer, pnp } }
}
